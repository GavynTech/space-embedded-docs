# Orbit Propagator

This page implements an embedded-friendly orbit propagator in C++. It uses a simplified J2-perturbed two-body model — accurate enough for onboard contact window prediction and eclipse detection without the memory and CPU cost of a full numerical integrator.

---

## Design Goals

- No dynamic memory allocation
- Fixed-point-safe (uses double, not float, for orbital accuracy)
- Runs in < 1 ms on a 200 MHz ARM Cortex-M7
- Predicts position/velocity for up to 24 hours ahead
- Computes eclipse and ground-station contact windows

## Data Structures

```cpp
// orbit_propagator.h
#pragma once
#include <cmath>
#include <cstdint>

namespace orb {

constexpr double MU_EARTH  = 3.986004418e14;   // m³/s²
constexpr double R_EARTH   = 6378137.0;         // m (equatorial)
constexpr double J2        = 1.08263e-3;
constexpr double OMEGA_E   = 7.2921150e-5;      // rad/s (Earth rotation)

struct Vec3 {
    double x, y, z;
    Vec3 operator+(const Vec3 &o) const { return {x+o.x, y+o.y, z+o.z}; }
    Vec3 operator-(const Vec3 &o) const { return {x-o.x, y-o.y, z-o.z}; }
    Vec3 operator*(double s)      const { return {x*s,   y*s,   z*s};   }
    double dot(const Vec3 &o)     const { return x*o.x + y*o.y + z*o.z; }
    double norm()                 const { return std::sqrt(dot(*this));  }
    Vec3   normalized()           const { double n=norm(); return {x/n, y/n, z/n}; }
};

struct OrbitalElements {
    double a;      // semi-major axis [m]
    double e;      // eccentricity
    double i;      // inclination [rad]
    double raan;   // right ascension of ascending node [rad]
    double aop;    // argument of perigee [rad]
    double m0;     // mean anomaly at epoch [rad]
    double epoch;  // TAI seconds since J2000
};

struct CartState {
    Vec3   pos;    // ECI position [m]
    Vec3   vel;    // ECI velocity [m/s]
    double t_tai;  // TAI time [s since J2000]
};

} // namespace orb
```

## Kepler Equation Solver

```cpp
// Solve M = E - e*sin(E) for E (eccentric anomaly)
static double solve_kepler(double M, double e) {
    double E = M;
    for (int iter = 0; iter < 50; iter++) {
        double dE = (M - E + e * std::sin(E)) / (1.0 - e * std::cos(E));
        E += dE;
        if (std::fabs(dE) < 1e-12) break;
    }
    return E;
}
```

## Elements to Cartesian

```cpp
namespace orb {

CartState elements_to_cart(const OrbitalElements &oe, double t_tai) {
    double dt = t_tai - oe.epoch;

    // J2-averaged secular drift rates
    double n    = std::sqrt(MU_EARTH / (oe.a * oe.a * oe.a));
    double p    = oe.a * (1.0 - oe.e * oe.e);
    double ci   = std::cos(oe.i);
    double n_J2 = n * (3.0/2.0) * J2 * (R_EARTH/p) * (R_EARTH/p);

    double raan_t = oe.raan - n_J2 * ci * dt;
    double aop_t  = oe.aop  + n_J2 * (2.5 * ci*ci - 0.5) * dt;
    double M_t    = oe.m0   + n * dt;

    // Solve Kepler's equation
    double E = solve_kepler(std::fmod(M_t, 2.0*M_PI), oe.e);

    // True anomaly
    double nu = 2.0 * std::atan2(
        std::sqrt(1.0 + oe.e) * std::sin(E/2.0),
        std::sqrt(1.0 - oe.e) * std::cos(E/2.0));

    // Position in perifocal frame
    double r = oe.a * (1.0 - oe.e * std::cos(E));
    Vec3 r_pf = { r * std::cos(nu), r * std::sin(nu), 0.0 };

    double h = std::sqrt(MU_EARTH * p);
    Vec3 v_pf = {
        -MU_EARTH / h * std::sin(nu),
         MU_EARTH / h * (oe.e + std::cos(nu)),
         0.0
    };

    // Rotate perifocal → ECI via 3-1-3 Euler: AOP, i, RAAN
    auto Rz = [](double a, Vec3 v) -> Vec3 {
        double c=std::cos(a), s=std::sin(a);
        return { c*v.x - s*v.y, s*v.x + c*v.y, v.z };
    };
    auto Rx = [](double a, Vec3 v) -> Vec3 {
        double c=std::cos(a), s=std::sin(a);
        return { v.x, c*v.y - s*v.z, s*v.y + c*v.z };
    };

    auto rotate = [&](Vec3 v) -> Vec3 {
        return Rz(raan_t, Rx(oe.i, Rz(aop_t, v)));
    };

    return { rotate(r_pf), rotate(v_pf), t_tai };
}

} // namespace orb
```

## Eclipse Detection

```cpp
namespace orb {

// Returns true if spacecraft is in Earth's shadow (cylindrical model)
bool in_eclipse(const Vec3 &pos_eci, const Vec3 &sun_eci) {
    Vec3 sun_hat = sun_eci.normalized();
    // Component of pos along sun direction
    double proj = pos_eci.dot(sun_hat);
    // Distance from sun-Earth line
    Vec3 perp = { pos_eci.x - proj*sun_hat.x,
                  pos_eci.y - proj*sun_hat.y,
                  pos_eci.z - proj*sun_hat.z };
    return (proj < 0.0) && (perp.norm() < R_EARTH);
}

} // namespace orb
```

## Contact Window Prediction

```cpp
namespace orb {

struct GroundStation {
    double lat_rad;
    double lon_rad;
    double alt_m;
    double min_elevation_rad;
};

// Check if spacecraft is visible from ground station at time t
bool is_visible(const CartState &sc, const GroundStation &gs, double gst_rad) {
    // ECEF position of ground station
    double cL = std::cos(gs.lat_rad), sL = std::sin(gs.lat_rad);
    double cl = std::cos(gs.lon_rad + gst_rad);
    double sl = std::sin(gs.lon_rad + gst_rad);
    double r_gs = R_EARTH + gs.alt_m;
    Vec3 gs_ecef = { r_gs*cL*cl, r_gs*cL*sl, r_gs*sL };

    // Convert ECI → ECEF
    Vec3 sc_ecef = {
         sc.pos.x*std::cos(gst_rad) + sc.pos.y*std::sin(gst_rad),
        -sc.pos.x*std::sin(gst_rad) + sc.pos.y*std::cos(gst_rad),
         sc.pos.z
    };

    Vec3 range = { sc_ecef.x - gs_ecef.x,
                   sc_ecef.y - gs_ecef.y,
                   sc_ecef.z - gs_ecef.z };

    // Elevation angle
    Vec3 gs_hat = gs_ecef.normalized();
    double el = std::asin(range.dot(gs_hat) / range.norm());
    return el >= gs.min_elevation_rad;
}

} // namespace orb
```

## Usage

```cpp
// Initialize from TLE-derived elements
orb::OrbitalElements iss = {
    .a    = 6371e3 + 420e3,   // 420 km altitude
    .e    = 0.0001234,
    .i    = 51.6 * M_PI/180,
    .raan = 210.3 * M_PI/180,
    .aop  = 89.5  * M_PI/180,
    .m0   = 270.5 * M_PI/180,
    .epoch = tai_j2000_seconds(2024, 4, 30, 12, 0, 0)
};

// Propagate 1 hour ahead in 10-second steps
for (double dt = 0; dt <= 3600.0; dt += 10.0) {
    orb::CartState state = orb::elements_to_cart(iss, iss.epoch + dt);
    // state.pos is ECI position [m], state.vel is ECI velocity [m/s]
}
```
